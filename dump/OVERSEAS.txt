CREATE  TABLE  OVERSEAS
(
   VOTER_ID  INTEGER(8),
  COUNTRY  VARCHAR(25) NOT NULL,
  PASSPORT_NO  INTEGER(10) NOT NULL,
  PASS_DATE  DATE NOT NULL,
  VISA_NO  INTEGER(10) NOT NULL,
  VISA_DATE  DATE NOT NULL,
  ADDRESS  VARCHAR(100),
  CONSTRAINT   OVERSEAS_ID_PK PRIMARY KEY(VOTER_ID),
  CONSTRAINT   OVERSEAS_ID_FK FOREIGN  KEY(VOTER_ID) REFERENCES VOTERS(ID) ON DELETE CASCADE,
  CONSTRAINT   OVERSEAS_COUNTRY_FK  FOREIGN  KEY(COUNTRY) REFERENCES  COUNTRIES(NAME) ON DELETE CASCADE 
);


DELIMITER $$
CREATE  PROCEDURE  OVERSEAS_INSERT  
 (
   IN ID  INTEGER(8),
   IN  COUNTRY  VARCHAR(25) ,
   IN  PNO  INTEGER(10),
   IN  PDATE  DATE,
   IN  VNO INTEGER(10),
   IN VDATE  DATE,
   IN  ADDR VARCHAR(100)
 )
BEGIN 
      INSERT  INTO   OVERSEAS(VOTER_ID,COUNTRY,PASSPORT_NO,PASS_DATE,VISA_NO,VISA_DATE,ADDRESS)VALUES(ID,COUNTRY,PNO,PDATE,VNO,VDATE,ADDR);

END  $$

DELIMITER $$
CREATE  PROCEDURE  OVERSEAS_UPDATE  
 (
   IN ID  INTEGER(8),
   IN  CONTRY  VARCHAR(25) ,
   IN  PNO  INTEGER(10),
   IN  PDATE  DATE,
   IN  VNO INTEGER(10),
   IN VDATE  DATE,
   IN  ADDR VARCHAR(100)
 )
BEGIN 

     UPDATE  OVERSEAS  SET VOTER_ID=ID,COUNTRY=CONTRY,PASSPORT_NO=PNO,PASS_DATE=PDATE,VISA_NO=VNO,VISA_DATE=VDATE,ADDRESS=ADDR;
END  $$
